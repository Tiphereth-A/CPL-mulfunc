\begin{frame}{Dirichlet 前缀和}
	\label{mulf:algo:dirichlet-psum}

	\only<1->{给出 \(a_{1},a_{2},\dots,a_{n}\), 求 \(b_{1},b_{2},\dots,b_{n}\), 其中

	\[
		b_i=\sum_{j\mid i}a_j
	\]}

	\only<2->{设 \(x,y\) 的分解分别为
		\[
			x=\prod_ip_i^{\alpha_i}, y=\prod_ip_i^{\beta_i}
		\]}

	\only<3->{则 \(a_x\) 对 \(b_y\) 有贡献当且仅当\(\forall i,\alpha_i\leqslant\beta_i\)}

	\only<4->{预先筛出 \([1,n]\) 中的素数即可更快地处理}

	\only<5->{\textbf{时间复杂度} 分析方法和 Eratosthenes 筛法一致, 为\(O(n\log\log n)\)}
\end{frame}
