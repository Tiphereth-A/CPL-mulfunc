\begin{frame}[fragile]{定义}
	\label{mulf:ssec:def}

	\only<1-3>{\begin{definition}[数论函数]
			\label{mulf:def:numf}

			对任意交换环 \(A\), 称函数 \(f:\mathbb{N}\to A\) 为数论函数
		\end{definition}}

	\only<2-3>{通俗来说就是满足如下要求的函数:
		\begin{itemize}
			\item 定义域为正整数
			\item 值域中任意两个元素的和, 差, 积均在值域中
			\item 值域上的和与积均具有交换律和结合律, 和与积之间具有分配律
			\item 值域中有 0 和 1
		\end{itemize}}

	\only<3>{如果某个数论函数具有简单的表达式\(f(n)\), 在不引起歧义的情况下我们也可以将其简写为\(\{f(n)\}\)

		如\(\{1\}\), \(\{n\}\)}

	\only<4-5>{\begin{definition}[积性函数]
			\label{mulf:def:mulf}

			满足
			\[
				\forall m, n\in\mathbb{N}^+,~(m,n)=1\implies f(mn)=f(m)f(n)
			\]

			的数论函数 \(f(n)\)
		\end{definition}}

	\only<5>{\begin{definition}[加性函数]
			\label{mulf:def:addf}

			满足

			\[
				\forall m, n\in\mathbb{N}^+,~(m,n)=1\implies f(mn)=f(m)+f(n)
			\]

			的数论函数 \(f(n)\)
		\end{definition}}

	\only<6>{\begin{definition}[完全积性函数]
			\label{mulf:def:cmulf}

			满足

			\[
				\forall m, n\in\mathbb{N}^+,~f(mn)=f(m)f(n)
			\]

			的 \(f(n)\) 的数论函数 \(f(n)\)
		\end{definition}}

	\only<7->{\begin{definition}[Dirichlet 卷积]
			\label{mulf:def:conv}

			若 \(f,g\) 为数论函数, 则称

			\begin{equation}
				\label{mulf:eq:conv}
				(f*g)(n):=\sum_{d\mid n}f(d)g\left(\frac{n}{d}\right)
			\end{equation}

			为 \(f,g\) 的 Dirichlet 卷积
		\end{definition}}

	\only<8->{如
		\[
			(f*g)(10)=f(1)g(10)+f(2)g(5)+f(5)g(2)+f(10)g(1)
		\]}
\end{frame}
